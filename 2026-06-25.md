# Python实现JSON Schema验证器

使用`jsonschema`库快速实现JSON数据验证，支持自定义错误提示，适合数据接口校验和配置检查场景。

```python
from typing import Any
from jsonschema import validate, ValidationError, Draft202012Validator
from json import dumps


class JsonSchemaValidator:
    """JSON Schema验证器，支持Draft 2020-12标准。
    
    Args:
        schema: JSON Schema字典，定义数据验证规则
        
    Example:
        >>> validator = JsonSchemaValidator({"type": "object", "properties": {"name": {"type": "string"}}})
        >>> validator.validate({"name": "Alice"})
        (True, "验证通过")
        >>> validator.validate({"name": 123})
        (False, "name: 123 is not of type 'string'")
    """
    
    def __init__(self, schema: dict[str, Any]) -> None:
        self._schema = schema
        Draft202012Validator.check_schema(schema)  # 验证Schema自身合法性
    
    def validate(self, data: Any) -> tuple[bool, str]:
        """验证数据是否符合Schema规范。
        
        Args:
            data: 待验证的JSON数据
            
        Returns:
            包含验证结果和错误信息的元组 (is_valid, message)
        """
        try:
            validate(instance=data, schema=self._schema)
            return (True, "验证通过")
        except ValidationError as e:
            error_path = ".".join(str(p) for p in e.absolute_path) or "根节点"
            error_msg = f"{error_path}: {e.message}" if e.absolute_path else e.message
            return (False, error_msg)


# 使用示例
if __name__ == "__main__":
    # 定义用户数据Schema
    user_schema = {
        "type": "object",
        "properties": {
            "name": {"type": "string", "minLength": 2},
            "age": {"type": "integer", "minimum": 0, "maximum": 150},
            "email": {"type": "string", "format": "email"},
            "tags": {"type": "array", "items": {"type": "string"}, "uniqueItems": True}
        },
        "required": ["name", "age", "email"],
        "additionalProperties": False
    }
    
    validator = JsonSchemaValidator(user_schema)
    
    # 测试数据1：符合规范
    valid_data = {
        "name": "张三",
        "age": 25,
        "email": "zhangsan@example.com",
        "tags": ["python", "developer"]
    }
    
    # 测试数据2：不符合规范
    invalid_data = {
        "name": "李",        # 长度不足
        "age": -5,           # 年龄非法
        "email": "invalid",  # 邮箱格式错误
        "extra_field": True  # 多余字段
    }
    
    print("=== 验证有效数据 ===")
    is_valid, message = validator.validate(valid_data)
    print(f"验证结果: {is_valid}")
    print(f"详细信息: {message}")
    
    print("\n=== 验证无效数据 ===")
    is_valid, message = validator.validate(invalid_data)
    print(f"验证结果: {is_valid}")
    print(f"详细信息: {message}")
```
