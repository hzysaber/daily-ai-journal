# Python多模型API统一封装

使用策略模式封装不同模型API，统一接口便于切换和扩展，支持流式输出和参数管理。

```python
import os
from typing import Protocol, Generator, Any, runtime_checkable
from dataclasses import dataclass, field
import requests


@runtime_checkable
class ModelAPI(Protocol):
    """统一模型接口协议"""
    def generate(
        self, 
        prompt: str, 
        stream: bool = False,
        **kwargs
    ) -> str | Generator[str, None, None]:
        """生成文本"""
        ...


@dataclass
class OpenAIAPI:
    """OpenAI API封装"""
    api_key: str = field(default_factory=lambda: os.getenv("OPENAI_API_KEY", ""))
    base_url: str = "https://api.openai.com/v1/chat/completions"
    
    def generate(
        self, 
        prompt: str, 
        stream: bool = False,
        model: str = "gpt-3.5-turbo",
        **kwargs
    ) -> str | Generator[str, None, None]:
        headers = {"Authorization": f"Bearer {self.api_key}"}
        data = {
            "model": model,
            "messages": [{"role": "user", "content": prompt}],
            "stream": stream,
            **kwargs
        }
        
        if stream:
            return self._stream_generate(headers, data)
        return self._normal_generate(headers, data)
    
    def _normal_generate(self, headers: dict, data: dict) -> str:
        response = requests.post(self.base_url, json=data, headers=headers)
        response.raise_for_status()
        return response.json()["choices"][0]["message"]["content"]
    
    def _stream_generate(self, headers: dict, data: dict) -> Generator[str, None, None]:
        response = requests.post(self.base_url, json=data, headers=headers, stream=True)
        response.raise_for_status()
        for line in response.iter_lines():
            if line:
                chunk = line.decode("utf-8").removeprefix("data: ")
                if chunk != "[DONE]":
                    yield chunk


@dataclass
class ClaudeAPI:
    """Claude API封装"""
    api_key: str = field(default_factory=lambda: os.getenv("ANTHROPIC_API_KEY", ""))
    base_url: str = "https://api.anthropic.com/v1/messages"
    
    def generate(
        self, 
        prompt: str, 
        stream: bool = False,
        model: str = "claude-3-sonnet-20240229",
        max_tokens: int = 1024,
        **kwargs
    ) -> str | Generator[str, None, None]:
        headers = {
            "x-api-key": self.api_key,
            "anthropic-version": "2023-06-01",
            "content-type": "application/json"
        }
        data = {
            "model": model,
            "max_tokens": max_tokens,
            "messages": [{"role": "user", "content": prompt}],
            "stream": stream,
            **kwargs
        }
        
        if stream:
            return self._stream_generate(headers, data)
        return self._normal_generate(headers, data)
    
    def _normal_generate(self, headers: dict, data: dict) -> str:
        response = requests.post(self.base_url, json=data, headers=headers)
        response.raise_for_status()
        return response.json()["content"][0]["text"]
    
    def _stream_generate(self, headers: dict, data: dict) -> Generator[str, None, None]:
        response = requests.post(self.base_url, json=data, headers=headers, stream=True)
        response.raise_for_status()
        for line in response.iter_lines():
            if line:
                chunk = line.decode("utf-8").removeprefix("data: ")
                if "content_block_delta" in chunk:
                    yield chunk


def get_model(model_name: str) -> ModelAPI:
    """工厂函数：获取模型实例"""
    models = {
        "openai": OpenAIAPI,
        "claude": ClaudeAPI
    }
    if model_name not in models:
        raise ValueError(f"不支持的模型：{model_name}，可选：{list(models.keys())}")
    return models[model_name]()
```

## 使用示例

```python
# 正常调用
openai_model = get_model("openai")
response = openai_model.generate("用Python写一个快速排序算法", model="gpt-4")
print(response)

# 流式调用
claude_model = get_model("claude")
stream = claude_model.generate("解释量子计算的基本原理", stream=True)
for chunk in stream:
    print(chunk, end="", flush=True)
```

## 说明

这个封装通过策略模式统一了不同模型API的调用接口，支持文本生成和流式输出两种模式。每个API类实现了相同的`generate`方法，通过工厂函数可以根据名称快速切换模型。使用时只需提供对应的API密钥，即可在OpenAI和Claude等模型间无缝切换。
